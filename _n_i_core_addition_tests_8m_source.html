00074 - (void)testNSString_isWhitespace {
00075   // From the Apple docs:
00076   // Returns a character set containing only the whitespace characters space (U+0020) and tab
00077   // (U+0009) and the newline and nextline characters (U+000Aâ€“U+000D, U+0085).
00078   STAssertTrue([@"" isWhitespaceAndNewlines], @"Empty string should be whitespace.");
00079   STAssertTrue([@" " isWhitespaceAndNewlines], @"Space character should be whitespace.");
00080   STAssertTrue([@"\t" isWhitespaceAndNewlines], @"Tab character should be whitespace.");
00081   STAssertTrue([@"\n" isWhitespaceAndNewlines], @"Newline character should be whitespace.");
00082   STAssertTrue([@"\r" isWhitespaceAndNewlines], @"Carriage return character should be whitespace.");
00083 
00084   // Unicode whitespace
00085   for (int unicode = 0x000A; unicode <= 0x000D; ++unicode) {
00086     NSString* str = [NSString stringWithFormat:@"%C", unicode];
00087     STAssertTrue([str isWhitespaceAndNewlines],
00088                  @"Unicode string #%X should be whitespace.", unicode);
00089   }
00090 
00091   NSString* str = [NSString stringWithFormat:@"%C", 0x0085];
00092   STAssertTrue([str isWhitespaceAndNewlines], @"Unicode string should be whitespace.");
00093 
00094   STAssertTrue([@" \t\r\n" isWhitespaceAndNewlines], @"Empty string should be whitespace.");
00095 
00096   STAssertTrue(![@"a" isWhitespaceAndNewlines], @"Text should not be whitespace.");
00097   STAssertTrue(![@" \r\n\ta\r\n " isWhitespaceAndNewlines], @"Text should not be whitespace.");
00098 }

00177 - (void)testNSString_stringByAddingQueryDictionary {
00178   NSString* baseUrl = @"http://google.com/search";
00179   STAssertTrue([[baseUrl stringByAddingQueryDictionary:nil] isEqualToString:
00180                 [baseUrl stringByAppendingString:@"?"]], @"Empty dictionary fail.");
00181 
00182   STAssertTrue([[baseUrl stringByAddingQueryDictionary:[NSDictionary dictionary]] isEqualToString:
00183                 [baseUrl stringByAppendingString:@"?"]], @"Empty dictionary fail.");
00184 
00185   STAssertTrue([[baseUrl stringByAddingQueryDictionary:[NSDictionary
00186                                                         dictionaryWithObject:@"three20"
00187                                                         forKey:@"q"]] isEqualToString:
00188                 [baseUrl stringByAppendingString:@"?q=three20"]], @"Single parameter fail.");
00189 
00190   NSDictionary* query = [NSDictionary
00191                          dictionaryWithObjectsAndKeys:
00192                          @"three20", @"q",
00193                          @"en",      @"hl",
00194                          nil];
00195   NSString* baseUrlWithQuery = [baseUrl stringByAddingQueryDictionary:query];
00196   STAssertTrue([baseUrlWithQuery isEqualToString:[baseUrl
00197                                                   stringByAppendingString:@"?hl=en&q=three20"]]
00198                || [baseUrlWithQuery isEqualToString:[baseUrl
00199                                                      stringByAppendingString:@"?q=three20&hl=en"]],
00200                @"Additional query parameters not correct. %@",
00201                [baseUrl stringByAddingQueryDictionary:query]);
