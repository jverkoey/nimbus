00074 // Builds a map of significant scopes to full scopes.
00075 //
00076 // For example, consider the following rulesets:
00077 //
00078 // .root UIButton {
00079 // }
00080 // .apple UIButton {
00081 // }
00082 // UIButton UIView {
00083 // }
00084 // UIView {
00085 // }
00086 //
00087 // The generated scope map will look like:
00088 //
00089 // UIButton => (.root UIButton, .apple UIButton)
00090 // UIView => (UIButton UIView, UIView)
00091 //
00092 - (void)rebuildSignificantScopeToScopes {
00093   NSMutableDictionary* significantScopeToScopes =
00094   [[NSMutableDictionary alloc] initWithCapacity:[_rawRulesets count]];
00095 
00096   for (NSString* scope in _rawRulesets) {
00097     NSArray* parts = [scope componentsSeparatedByString:@" "];
00098     NSString* mostSignificantScopePart = [parts lastObject];
00099 
00100     // TODO (jverkoey Oct 6, 2011): We should respect CSS specificity. Right now this will
00101     // give higher precedance to newer styles. Instead, we should prefer styles that have more
00102     // selectors.
00103     NSMutableArray* scopes = [significantScopeToScopes objectForKey:mostSignificantScopePart];
00104     if (nil == scopes) {
00105       scopes = [[NSMutableArray alloc] initWithObjects:scope, nil];
00106       [significantScopeToScopes setObject:scopes forKey:mostSignificantScopePart];
00107       NI_RELEASE_SAFELY(scopes);
00108       
00109     } else {
00110       [scopes addObject:scope];
00111     }
00112   }
00113 
00114   [_significantScopeToScopes release];
00115   _significantScopeToScopes = [significantScopeToScopes copy];
00116   
00117   NI_RELEASE_SAFELY(significantScopeToScopes);
00118 }

00196 - (void)addStylesheet:(NIStylesheet *)stylesheet {
00197   NIDASSERT(nil != stylesheet);
00198   if (nil == stylesheet) {
00199     return;
00200   }
00201 
00202   @synchronized(self) {
00203     NSMutableDictionary* compositeRuleSets = [self.rawRulesets mutableCopy];
00204 
00205     BOOL ruleSetsDidChange = NO;
00206 
00207     for (NSString* selector in stylesheet.rawRulesets) {
00208       NSDictionary* incomingRuleSet   = [stylesheet.rawRulesets objectForKey:selector];
00209       NSDictionary* existingRuleSet = [self.rawRulesets objectForKey:selector];
00210 
00211       // Don't bother adding empty rulesets.
00212       if ([incomingRuleSet count] > 0) {
00213         ruleSetsDidChange = YES;
00214 
00215         if (nil == existingRuleSet) {
00216           // There is no rule set of this selector - simply add the new one.
00217           [compositeRuleSets setObject:incomingRuleSet forKey:selector];
00218           continue;
00219         }
00220 
00221         NSMutableDictionary* compositeRuleSet = [existingRuleSet mutableCopy];
00222         // Add the incoming rule set entries, overwriting any existing ones.
00223         [compositeRuleSet addEntriesFromDictionary:incomingRuleSet];
00224 
00225         [compositeRuleSets setObject:compositeRuleSet forKey:selector];
00226         NI_RELEASE_SAFELY(compositeRuleSet);
00227       }
00228     }
00229 
00230     NI_RELEASE_SAFELY(_rawRulesets);
00231     _rawRulesets = [compositeRuleSets copy];
00232     NI_RELEASE_SAFELY(compositeRuleSets);
00233 
00234     if (ruleSetsDidChange) {
00235       [self ruleSetsDidChange];
00236     }
00237   }
00238 }

00255 - (void)applyStyleToView:(UIView *)view withClassName:(NSString *)className {
00256   NSArray* selectors = [_significantScopeToScopes objectForKey:className];
00257   if ([selectors count] > 0) {
00258     // Gather all of the rule sets for this view into a composite rule set.
00259     NICSSRuleset* ruleSet = [_ruleSets objectForKey:className];
00260 
00261     if (nil == ruleSet) {
00262       ruleSet = [[NICSSRuleset alloc] init];
00263 
00264       // Composite the rule sets into one.
00265       for (NSString* selector in selectors) {
00266         [ruleSet addEntriesFromDictionary:[_rawRulesets objectForKey:selector]];
00267       }
00268 
00269       NIDASSERT(nil != _ruleSets);
00270       [_ruleSets setObject:ruleSet forKey:className];
00271       [ruleSet release]; // We can release the ruleSet because it's retained by the dictionary.
00272     }
00273 
00274     [self applyRuleSet:ruleSet toView:view];
00275   }
