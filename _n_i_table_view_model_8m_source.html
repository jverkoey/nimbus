00140   [self _resetCompiledData];
00141

00143 
00144   NSString* currentSectionHeaderTitle = nil;
00145   NSString* currentSectionFooterTitle = nil;

00150     BOOL isSectionFooter = [object isKindOfClass:[NITableViewModelFooter class]];
00151

00156 
00157     } else if (isSectionFooter) {

00159       currentSectionFooterTitle = footer.title;
00160 
00161     } else {
00162       if (nil == currentSectionRows) {
00163         currentSectionRows = [[NSMutableArray alloc] init];
00164       }
00165       [currentSectionRows addObject:object];
00166     }
00167 
00168     // A section footer or title has been encountered,
00169     if (nil != nextSectionHeaderTitle || nil != currentSectionFooterTitle) {
00170       if (nil != currentSectionHeaderTitle
00171           || nil != currentSectionFooterTitle
00172           || nil != currentSectionRows) {
00173         NITableViewModelSection* section = [NITableViewModelSection section];
00174         section.headerTitle = currentSectionHeaderTitle;
00175         section.footerTitle = currentSectionFooterTitle;
00176         section.rows = currentSectionRows;
00177         [sections addObject:section];
00178       }
00179 
00180       NI_RELEASE_SAFELY(currentSectionRows);
00181       currentSectionHeaderTitle = nextSectionHeaderTitle;
00182       currentSectionFooterTitle = nil;
00183     }
00184   }
00185 
00186   // Commit any unfinished sections.
00187   if ([currentSectionRows count] > 0 || nil != currentSectionHeaderTitle) {
00188     NITableViewModelSection* section = [NITableViewModelSection section];
00189     section.headerTitle = currentSectionHeaderTitle;

00202 - (void)_compileSectionIndex {
00203   NI_RELEASE_SAFELY(_sectionIndexTitles);
00204 
00205   // Prime the section index and the map
00206   NSMutableArray* titles = nil;
00207   NSMutableDictionary* sectionPrefixToSectionIndex = nil;
00208   if (NITableViewModelSectionIndexNone != _sectionIndexType) {
00209     titles = [NSMutableArray array];
00210     sectionPrefixToSectionIndex = [NSMutableDictionary dictionary];
00211 
00212     // The search symbol is always first in the index.
00213     if (_sectionIndexShowsSearch) {
00214       [titles addObject:UITableViewIndexSearch];
00215     }
00216   }
00217 
00218   // A dynamic index shows the first letter of every section in the index in whatever order the
00219   // sections are ordered (this may not be alphabetical).
00220   if (NITableViewModelSectionIndexDynamic == _sectionIndexType) {
00221     for (NITableViewModelSection* section in _sections) {
00222       NSString* headerTitle = section.headerTitle;
00223       if ([headerTitle length] > 0) {
00224         NSString* prefix = [headerTitle substringToIndex:1];
00225         [titles addObject:prefix];
00226       }
00227     }
00228 
00229   } else if (NITableViewModelSectionIndexAlphabetical == _sectionIndexType) {
00230     // Use the localized indexed collation to create the index. In English, this will always be
00231     // the entire alphabet.
00232     NSArray* sectionIndexTitles = [[UILocalizedIndexedCollation currentCollation] sectionIndexTitles];
00233 
00234     // The localized indexed collection sometimes includes a # for summaries, but we might
00235     // not want to show a summary in the index, so prune it out. It's not guaranteed that
00236     // a # will actually be included in the section index titles, so we always attempt to
00237     // remove it for consistency's sake and then add it back down below if it is requested.
00238     for (NSString* letter in sectionIndexTitles) {
00239       if (![letter isEqualToString:@"#"]) {
00240         [titles addObject:letter];
00241       }
00242     }
00243   }
00244 
00245   // Add the section summary symbol if it was requested.
00246   if (_sectionIndexShowsSummary) {
00247     [titles addObject:@"#"];
00248   }
00249 
00250   // Build the prefix => section index map.
00251   if (NITableViewModelSectionIndexNone != _sectionIndexType) {
00252 
00253     // Map all of the sections to indices.
00254     NSInteger sectionIndex = 0;
00255     for (NITableViewModelSection* section in _sections) {
00256       NSString* headerTitle = section.headerTitle;
00257       if ([headerTitle length] > 0) {
00258         NSString* prefix = [headerTitle substringToIndex:1];
00259         if (nil == [sectionPrefixToSectionIndex objectForKey:prefix]) {
00260           [sectionPrefixToSectionIndex setObject:[NSNumber numberWithInt:sectionIndex] forKey:prefix];
00261         }
00262       }
00263       ++sectionIndex;
00264     }
00265 
00266     // Map the unmapped section titles to the next closest earlier section.
00267     NSInteger lastIndex = 0;
00268     for (NSString* title in titles) {
00269       NSString* prefix = [title substringToIndex:1];
00270       if (nil != [sectionPrefixToSectionIndex objectForKey:prefix]) {
00271         lastIndex = [[sectionPrefixToSectionIndex objectForKey:prefix] intValue];
00272         
00273       } else {
00274         [sectionPrefixToSectionIndex setObject:[NSNumber numberWithInt:lastIndex] forKey:prefix];
00275       }
00276     }
00277   }
00278 
00279   _sectionIndexTitles = [titles copy];
00280   _sectionPrefixToSectionIndex = [sectionPrefixToSectionIndex copy];
