00195 - (void)_compileSectionIndex {
00196   NI_RELEASE_SAFELY(_sectionIndexTitles);
00197 
00198   // Prime the section index and the map
00199   NSMutableArray* titles = nil;
00200   NSMutableDictionary* sectionPrefixToSectionIndex = nil;
00201   if (NITableViewModelSectionIndexNone != _sectionIndexType) {
00202     titles = [NSMutableArray array];
00203     sectionPrefixToSectionIndex = [NSMutableDictionary dictionary];
00204 
00205     // The search symbol is always first in the index.
00206     if (_sectionIndexShowsSearch) {
00207       [titles addObject:UITableViewIndexSearch];
00208     }
00209   }
00210 
00211   // A dynamic index shows the first letter of every section in the index in whatever order the
00212   // sections are ordered (this may not be alphabetical).
00213   if (NITableViewModelSectionIndexDynamic == _sectionIndexType) {
00214     for (NITableViewModelSection* section in _sections) {
00215       NSString* headerTitle = section.headerTitle;
00216       if ([headerTitle length] > 0) {
00217         NSString* prefix = [headerTitle substringToIndex:1];
00218         [titles addObject:prefix];
00219       }
00220     }
00221 
00222     // Use the localized indexed collation to create the index. In English, this will always be
00223     // the entire alphabet.
00224   } else if (NITableViewModelSectionIndexAlphabetical == _sectionIndexType) {
00225     NSArray* sectionIndexTitles = [[UILocalizedIndexedCollation currentCollation] sectionIndexTitles];
00226 
00227     // The localized indexed collection sometimes includes a # for summaries, but we might
00228     // not want to show a summary in the index, so prune it out. It's not guaranteed that
00229     // a # will actually be included in the section index titles, so we always attempt to
00230     // remove it for consistency's sake and then add it back down below if it is requested.
00231     for (NSString* letter in sectionIndexTitles) {
00232       if (![letter isEqualToString:@"#"]) {
00233         [titles addObject:letter];
00234       }
00235     }
00236   }
00237 
00238   // Add the section summary symbol if it was requested.
00239   if (_sectionIndexShowsSummary) {
00240     [titles addObject:@"#"];
00241   }
00242 
00243   // Build the prefix => section index map.
00244   if (NITableViewModelSectionIndexNone != _sectionIndexType) {
00245 
00246     // Map all of the sections to indices.
00247     NSInteger sectionIndex = 0;
00248     for (NITableViewModelSection* section in _sections) {
00249       NSString* headerTitle = section.headerTitle;
00250       if ([headerTitle length] > 0) {
00251         NSString* prefix = [headerTitle substringToIndex:1];
00252         if (nil == [sectionPrefixToSectionIndex objectForKey:prefix]) {
00253           [sectionPrefixToSectionIndex setObject:[NSNumber numberWithInt:sectionIndex] forKey:prefix];
00254         }
00255       }
00256       ++sectionIndex;
00257     }
00258 
00259     // Map the unmapped section titles to the next closest earlier section.
00260     NSInteger lastIndex = 0;
00261     for (NSString* title in titles) {
00262       NSString* prefix = [title substringToIndex:1];
00263       if (nil != [sectionPrefixToSectionIndex objectForKey:prefix]) {
00264         lastIndex = [[sectionPrefixToSectionIndex objectForKey:prefix] intValue];
00265         
00266       } else {
00267         [sectionPrefixToSectionIndex setObject:[NSNumber numberWithInt:lastIndex] forKey:prefix];
00268       }
00269     }
00270   }
00271 
00272   _sectionIndexTitles = [titles copy];
00273   _sectionPrefixToSectionIndex = [sectionPrefixToSectionIndex copy];
00274 }
