00115 
00116   // Don't animate the overview if the device didn't actually change orientations.
00117   if (lastOrientation != [[UIDevice currentDevice] orientation]
00118       && [[UIDevice currentDevice] orientation] != UIDeviceOrientationUnknown
00119       && [[UIDevice currentDevice] orientation] != UIDeviceOrientationFaceUp
00120       && [[UIDevice currentDevice] orientation] != UIDeviceOrientationFaceDown) {
00121 
00122     // When we flip from landscape to landscape or portait to portait the animation lasts
00123     // twice as long.
00124     UIDeviceOrientation currentOrientation = [[UIDevice currentDevice] orientation];
00125     BOOL isLongAnimation = (UIDeviceOrientationIsLandscape(lastOrientation)
00126                             == UIDeviceOrientationIsLandscape(currentOrientation));
00127     lastOrientation = currentOrientation;
00128 
00129     // Hide the overview right away, we'll make it fade back in when the rotation is
00130     // finished.

00134     [self cancelPreviousPerformRequestsWithTarget:self];

00136                afterDelay:NIDeviceRotationDuration(isLongAnimation)];

00320 
00321   } else if (orient == UIInterfaceOrientationLandscapeRight) {
00322     overviewWidth = [UIScreen mainScreen].bounds.size.height;
00323     frame = CGRectMake([UIScreen mainScreen].bounds.size.width
00324                        - (NIOverviewStatusBarHeight() + sOverviewHeight), 0,
00325                        sOverviewHeight, overviewWidth);
00326 
00327   } else if (orient == UIInterfaceOrientationPortraitUpsideDown) {
00328     overviewWidth = [UIScreen mainScreen].bounds.size.width;
00329     frame = CGRectMake(0, [UIScreen mainScreen].bounds.size.height
00330                        - (NIOverviewStatusBarHeight() + sOverviewHeight),
00331                        overviewWidth, sOverviewHeight);
00332 
00333   } else if (orient == UIInterfaceOrientationPortrait) {
00334     overviewWidth = [UIScreen mainScreen].bounds.size.width;
00335     frame = CGRectMake(0, NIOverviewStatusBarHeight(), overviewWidth, sOverviewHeight);
00336 
00337   } else {
00338     overviewWidth = [UIScreen mainScreen].bounds.size.width;
00339     frame = CGRectMake(0, NIOverviewStatusBarHeight(), overviewWidth, sOverviewHeight);
00340   }
00341 
00342   if ([[UIApplication sharedApplication] isStatusBarHidden]) {
00343     // When the status bar is hidden we want to position the overview offscreen.
00344     switch (orient) {
00345       case UIInterfaceOrientationLandscapeLeft: {
00346         frame = CGRectOffset(frame, -frame.size.width, 0);
00347         break;
00348       }
00349       case UIInterfaceOrientationLandscapeRight: {
00350         frame = CGRectOffset(frame, frame.size.width, 0);
00351         break;
00352       }
00353       case UIInterfaceOrientationPortrait: {
00354         frame = CGRectOffset(frame, 0, -frame.size.height);
00355         break;
00356       }
