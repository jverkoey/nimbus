00248     state->itemsPtr = stackbuf;

00251       // Initialize the state here instead of above when we check 0 == state->state because
00252       // for single item linked lists head == tail.

00255 
00256     while ((0 != state->state) && (numberOfItemsReturned < len)) {
00257       struct NILinkedListNode* node = (struct NILinkedListNode *)state->state;
00258       stackbuf[numberOfItemsReturned] = node->object;
