00042 - (void)drawGraphWithContext:(CGContextRef)context {
00043   CGSize contentSize = self.bounds.size;
00044 
00045   CGFloat xRange = [self.dataSource graphViewXRange:self];
00046   CGFloat yRange = [self.dataSource graphViewYRange:self];
00047 
00048   [self.dataSource resetPointIterator];
00049 
00050   CGContextSetLineWidth(context, 1);
00051   CGContextSetShouldAntialias(context, YES);
00052 
00053   BOOL isFirstPoint = YES;
00054   CGPoint point = CGPointZero;
00055   while ([self.dataSource nextPointInGraphView:self point:&point]) {
00056     CGPoint scaledPoint = CGPointMake(point.x / xRange, point.y / yRange);
00057     CGPoint plotPoint = CGPointMake(floorf(scaledPoint.x * contentSize.width) - 0.5f,
00058                                     contentSize.height
00059                                     - floorf((scaledPoint.y * 0.8f + 0.1f)
00060                                              * contentSize.height) - 0.5f);
00061     if (!isFirstPoint) {
00062       CGContextAddLineToPoint(context, plotPoint.x, plotPoint.y);
00063     }
00064     CGContextMoveToPoint(context, plotPoint.x, plotPoint.y);
00065     isFirstPoint = NO;
00066   }
00067 
00068   CGContextSetStrokeColorWithColor(context, [UIColor colorWithWhite:1 alpha:0.6f].CGColor);
00069   CGContextStrokePath(context);
00070   
00071   [self.dataSource resetEventIterator];
00072 
00073   CGFloat xValue = 0;
00074   UIColor* color = nil;
00075   while ([self.dataSource nextEventInGraphView:self xValue:&xValue color:&color]) {
00076     CGFloat scaledXValue = xValue / xRange;
00077     CGFloat plotXValue = floorf(scaledXValue * contentSize.width) - 0.5f;
00078     CGContextMoveToPoint(context, plotXValue, 0);
00079     CGContextAddLineToPoint(context, plotXValue, contentSize.height);
00080 
00081     CGContextSetStrokeColorWithColor(context, color.CGColor);
00082     CGContextStrokePath(context);
00083   }
00084 }

00108 
00109   colorspace = CGColorSpaceCreateDeviceRGB();
00110   glossGradient = CGGradientCreateWithColorComponents(colorspace,
00111                                                       components, locations, numberOfLocations);
00112 
00113   CGPoint topCenter = CGPointMake(CGRectGetMidX(bounds), 0.0f);
00114   CGPoint midCenter = CGPointMake(CGRectGetMidX(bounds), CGRectGetMidY(bounds));
00115   CGContextDrawLinearGradient(context, glossGradient, topCenter, midCenter, 0);

00119   CGColorSpaceRelease(colorspace);
