00174 - (CGRect)rectAroundPoint:(CGPoint)point atZoomScale:(CGFloat)zoomScale {
00175   NIDASSERT(zoomScale > 0);
00176 
00177   // Define the shape of the zoom rect.
00178   CGSize boundsSize = self.bounds.size;
00179 
00180   // Modify the size according to the requested zoom level.
00181   // For example, if we're zooming in to 0.5 zoom, then this will increase the bounds size
00182   // by a factor of two.
00183   CGSize scaledBoundsSize = CGSizeMake(boundsSize.width / zoomScale,
00184                                        boundsSize.height / zoomScale);
00185 
00186   CGRect rect = CGRectMake(point.x - scaledBoundsSize.width / 2,
00187                            point.y - scaledBoundsSize.height / 2,
00188                            scaledBoundsSize.width,
00189                            scaledBoundsSize.height);
00190 
00191   // When the image is zoomed out there is a bit of empty space around the image due
00192   // to the fact that it's centered on the screen. When we created the rect around the
00193   // point we need to take this "space" into account.
00194 
00195   // 1: get the frame of the image in this view's coordinates.
00196   CGRect imageScaledFrame = [self convertRect:_imageView.frame toView:self];
00197 
00198   // 2: Offset the frame by the excess amount. This will ensure that the zoomed location
00199   //    is always centered on the tap location. We only allow positive values because a
00200   //    negative value implies that there isn't actually any offset.
00201   rect = CGRectOffset(rect, -MAX(0, imageScaledFrame.origin.x), -MAX(0, imageScaledFrame.origin.y));
00202 
00203   return rect;
00204 }

00378 - (void)minAndMaxScaleForDimensions: (CGSize)dimensions
00379                          boundsSize: (CGSize)boundsSize
00380                           photoSize: (NIPhotoScrollViewPhotoSize)photoSize
00381                            minScale: (CGFloat *)pMinScale
00382                            maxScale: (CGFloat *)pMaxScale {
00383   NIDASSERT(nil != pMinScale);
00384   NIDASSERT(nil != pMaxScale);
00385   if (nil == pMinScale
00386       || nil == pMaxScale) {
00387     return;
00388   }
00389 
00390   CGFloat minScale = [self scaleForSize: dimensions
00391                              boundsSize: boundsSize
00392                         useMinimalScale: YES];
00393 
00394   // On high resolution screens we have double the pixel density, so we will be seeing
00395   // every pixel if we limit the maximum zoom scale to 0.5.
00396   // If the photo size is unknown, it's likely that we're showing the loading image and
00397   // don't want to shrink it down with the zoom because it should be a scaled image.
00398   CGFloat maxScale = ((NIPhotoScrollViewPhotoSizeUnknown == photoSize)
00399                       ? 1
00400                       : (1.0f / NIScreenScale()));
00401 
00402   if (NIPhotoScrollViewPhotoSizeThumbnail != photoSize) {
00403     // Don't let minScale exceed maxScale. (If the image is smaller than the screen, we
00404     // don't want to force it to be zoomed.)
00405     minScale = MIN(minScale, maxScale);
00406   }
00407 
00408   // At this point if the image is small, then minScale and maxScale will be the same because
00409   // we don't want to allow the photo to be zoomed.
00410 
00411   // If zooming above the original size IS enabled, however, expand the max zoom to
00412   // whatever value would make the image fit the view perfectly.
00413   if ([self isZoomingAboveOriginalSizeEnabled]) {
00414     CGFloat idealMaxScale = [self scaleForSize: dimensions
00415                                     boundsSize: boundsSize
00416                                useMinimalScale: NO];
00417     maxScale = MAX(maxScale, idealMaxScale);
00418   }
00419 
00420   *pMinScale = minScale;
00421   *pMaxScale = maxScale;

00426 - (void)setMaxMinZoomScalesForCurrentBounds {
00427   CGSize imageSize = _imageView.bounds.size;
00428   
00429   // Avoid crashing if the image has no dimensions.
00430   NIDASSERT(imageSize.width > 0 && imageSize.height > 0);
00431   if (imageSize.width <= 0 || imageSize.height <= 0) {
00432     _scrollView.maximumZoomScale = 1;
00433     _scrollView.minimumZoomScale = 1;
00434     return;
00435   }
00436   
00437   // The following code is from Apple's ImageScrollView example application and has been used
00438   // here because it is well-documented and concise.
00439   
00440   CGSize boundsSize = _scrollView.bounds.size;
00441   
00442   CGFloat minScale = 0;
00443   CGFloat maxScale = 0;
00444   
00445   // Calculate the min/max scale for the image to be presented.
00446   [self minAndMaxScaleForDimensions: imageSize
00447                          boundsSize: boundsSize
00448                           photoSize: self.photoSize
00449                            minScale: &minScale
00450                            maxScale: &maxScale];
00451   
00452   // When we show thumbnails for images that are too small for the bounds, we try to use
00453   // the known photo dimensions to scale the minimum scale to match what the final image
00454   // would be. This avoids any "snapping" effects from stretching the thumbnail too large.
00455   if ((NIPhotoScrollViewPhotoSizeThumbnail == self.photoSize)
00456       && !CGSizeEqualToSize(self.photoDimensions, CGSizeZero)) {
00457     CGFloat scaleToFitOriginal = 0;
00458     CGFloat originalMaxScale = 0;
00459     // Calculate the original-sized image's min/max scale.
00460     [self minAndMaxScaleForDimensions: self.photoDimensions
00461                            boundsSize: boundsSize
00462                             photoSize: NIPhotoScrollViewPhotoSizeOriginal
00463                              minScale: &scaleToFitOriginal
00464                              maxScale: &originalMaxScale];
00465     
00466     if (scaleToFitOriginal + FLT_EPSILON >= (1.0 / NIScreenScale())) {
00467       // If the final image will be smaller than the view then we want to use that
00468       // scale as the "true" scale and adjust it relatively to the thumbnail's dimensions.
00469       // This ensures that the thumbnail will always be the same visual size as the original
00470       // image, giving us that sexy "crisping" effect when the thumbnail is loaded.
00471       CGFloat relativeSize = self.photoDimensions.width / imageSize.width;
00472       minScale = scaleToFitOriginal * relativeSize;
00473     }
00474   }
00475   
00476   // If zooming is disabled then we flatten the range for zooming to only allow the min zoom.
00477   _scrollView.maximumZoomScale = [self isZoomingEnabled] ? maxScale : minScale;
00478   _scrollView.minimumZoomScale = minScale;
00479 }

00529 - (void)restoreCenterPoint:(CGPoint)oldCenter scale:(CGFloat)oldScale {
00530   // Step 1: restore zoom scale, making sure it is within the allowable range.
00531   _scrollView.zoomScale = boundf(oldScale,
00532                                  _scrollView.minimumZoomScale, _scrollView.maximumZoomScale);
00533 
00534   // Step 2: restore center point, making sure it is within the allowable range.
00535 
00536   // 2a: convert our desired center point back to the scroll view's coordinate space from the
00537   //     image's coordinate space.
00538   CGPoint boundsCenter = [self convertPoint:oldCenter fromView:_imageView];
00539 
00540   // 2b: calculate the content offset that would yield that center point
00541   CGPoint offset = CGPointMake(boundsCenter.x - _scrollView.bounds.size.width / 2.0f,
00542                                boundsCenter.y - _scrollView.bounds.size.height / 2.0f);
00543 
00544   // 2c: restore offset, adjusted to be within the allowable range
00545   CGPoint maxOffset = [self maximumContentOffset];
00546   CGPoint minOffset = [self minimumContentOffset];
00547   offset.x = boundf(offset.x, minOffset.x, maxOffset.x);
00548   offset.y = boundf(offset.y, minOffset.y, maxOffset.y);
00549   _scrollView.contentOffset = offset;
00550 }
