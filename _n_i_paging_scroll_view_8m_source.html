00126 - (CGRect)frameForPagingScrollView {
00127   CGRect frame = self.bounds;
00128 
00129   // We make the paging scroll view a little bit wider on the side edges so that there
00130   // there is space between the pages when flipping through them.
00131   frame.origin.x -= self.pageHorizontalMargin;
00132   frame.size.width += (2 * self.pageHorizontalMargin);
00133 
00134   return frame;

00139 - (CGRect)frameForPageAtIndex:(NSInteger)pageIndex {
00140   // We have to use our paging scroll view's bounds, not frame, to calculate the page
00141   // placement. When the device is in landscape orientation, the frame will still be in
00142   // portrait because the pagingScrollView is the root view controller's view, so its
00143   // frame is in window coordinate space, which is never rotated. Its bounds, however,
00144   // will be in landscape because it has a rotation transform applied.
00145   CGRect bounds = self.pagingScrollView.bounds;
00146   CGRect pageFrame = bounds;
00147 
00148   // We need to counter the extra spacing added to the paging scroll view in
00149   // frameForPagingScrollView:
00150   pageFrame.size.width -= self.pageHorizontalMargin * 2;
00151   pageFrame.origin.x = (bounds.size.width * pageIndex) + self.pageHorizontalMargin;
00152 
00153   return pageFrame;
00154 }

00276 - (void)updateVisiblePages {
00277   NSRange visiblePageRange = [self visiblePageRange];
00278 
00279   // Recycle no-longer-visible pages. We copy _visiblePages because we may modify it while we're
00280   // iterating over it.
00281   for (UIView<NIPagingScrollViewPage>* page in [[_visiblePages copy] autorelease]) {
00282     if (!NSLocationInRange(page.pageIndex, visiblePageRange)) {
00283       [_viewRecycler recycleView:page];
00284       [page removeFromSuperview];
00285 
00286       [self didRecyclePage:page];
00287 
00288       [_visiblePages removeObject:page];
00289     }
00290   }
00291 
00292   NSInteger oldCenterPageIndex = self.centerPageIndex;
00293     
00294   if (_numberOfPages > 0) {
00295     _centerPageIndex = [self currentVisiblePageIndex];
00296       
00297     // Prioritize displaying the currently visible page.
00298     if (![self isDisplayingPageForIndex:_centerPageIndex]) {
00299       [self displayPageAtIndex:_centerPageIndex];
00300     }
00301       
00302     // Add missing pages.
00303     for (int pageIndex = visiblePageRange.location;
00304          pageIndex < NSMaxRange(visiblePageRange); ++pageIndex) {
00305       if (![self isDisplayingPageForIndex:pageIndex]) {
00306         [self displayPageAtIndex:pageIndex];
00307       }
00308     }
00309   } else {
00310     _centerPageIndex = -1;
00311   }
00312 
00313   if (oldCenterPageIndex != _centerPageIndex
00314       && [self.delegate respondsToSelector:@selector(pagingScrollViewDidChangePages:)]) {
00315     [self.delegate pagingScrollViewDidChangePages:self];
00316   }
00317 }

00430 - (void)reloadData {
00431   NIDASSERT(nil != _dataSource);
00432 
00433   // Remove any visible pages from the view before we release the sets.
00434   for (UIView<NIPagingScrollViewPage>* page in _visiblePages) {
00435     [_viewRecycler recycleView:page];
00436     [(UIView *)page removeFromSuperview];
00437   }

00444     self.pagingScrollView.contentSize = self.bounds.size;
00445     self.pagingScrollView.contentOffset = CGPointZero;

00447 
00448     // May as well just get rid of all the views then.
00449     [_viewRecycler removeAllViews];

00451     return;
00452   }

00455

00457   _numberOfPages = [_dataSource numberOfPagesInPagingScrollView:self];

00466     _isModifyingContentOffset = YES;

00480                                 duration: (NSTimeInterval)duration {
00481   // Here, our pagingScrollView bounds have not yet been updated for the new interface
00482   // orientation. This is a good place to calculate the content offset that we will

00555     return;
