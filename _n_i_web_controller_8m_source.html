00088 - (void)shareAction {
00089   // Dismiss the action menu if the user taps the action button again on the iPad.
00090   if ([_actionSheet isVisible]) {
00091     // It shouldn't be possible to tap the share action button again on anything but the iPad.
00092     NIDASSERT(NIIsPad());
00093 
00094     [_actionSheet dismissWithClickedButtonIndex:[_actionSheet cancelButtonIndex] animated:YES];
00095 
00096     // We remove the action sheet here just in case the delegate isn't properly implemented.
00097     NI_RELEASE_SAFELY(_actionSheet);
00098     NI_RELEASE_SAFELY(_actionSheetURL);
00099 
00100     // Don't show the menu again.
00101     return;
00102   }
00103 
00104   // Remember the URL at this point
00105   [_actionSheetURL release];
00106   _actionSheetURL = [self.URL copy];
00107 
00108   if (nil == _actionSheet) {
00109     _actionSheet =
00110     [[UIActionSheet alloc] initWithTitle:[_actionSheetURL absoluteString]
00111                                 delegate:self
00112                        cancelButtonTitle:nil
00113                   destructiveButtonTitle:nil
00114                        otherButtonTitles:nil];
00115     // Let -shouldPresentActionSheet: setup the action sheet
00116     if (![self shouldPresentActionSheet:_actionSheet]) {
00117       // A subclass decided to handle the action in another way
00118       NI_RELEASE_SAFELY(_actionSheet);
00119       NI_RELEASE_SAFELY(_actionSheetURL);
00120       return;
00121     }
00122     // Add "Cancel" button except for iPads
00123     if (!NIIsPad()) {
00124       [_actionSheet setCancelButtonIndex:[_actionSheet addButtonWithTitle:NSLocalizedString(@"Cancel", @"")]];
00125     }
00126   }
00127 
00128   if (NIIsPad()) {
00129     [_actionSheet showFromBarButtonItem:_actionButton animated:YES];
00130 
00131   } else {
00132     [_actionSheet showInView:self.view];
00133   }
