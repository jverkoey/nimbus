00132 {
00133         if (![operation isKindOfClass:[ASIHTTPRequest class]]) {
00134                 [NSException raise:@"AttemptToAddInvalidRequest" format:@"Attempted to add an object that was not an ASIHTTPRequest to an ASINetworkQueue"];
00135         }

00138         
00139         ASIHTTPRequest *request = (ASIHTTPRequest *)operation;
00140         
00141         if ([self showAccurateProgress]) {
00142                 
00143                 // Force the request to build its body (this may change requestMethod)
00144                 [request buildPostBody];
00145                 
00146                 // If this is a GET request and we want accurate progress, perform a HEAD request first to get the content-length
00147                 // We'll only do this before the queue is started
00148                 // If requests are added after the queue is started they will probably move the overall progress backwards anyway, so there's no value performing the HEAD requests first
00149                 // Instead, they'll update the total progress if and when they receive a content-length header
00150                 if ([[request requestMethod] isEqualToString:@"GET"]) {
00151                         if ([self isSuspended]) {
00152                                 ASIHTTPRequest *HEADRequest = [request HEADRequest];
00153                                 [self addHEADOperation:HEADRequest];
00154                                 [request addDependency:HEADRequest];
00155                                 if ([request shouldResetDownloadProgress]) {
00156                                         [self resetProgressDelegate:&downloadProgressDelegate];
00157                                         [request setShouldResetDownloadProgress:NO];
00158                                 }
00159                         }
00160                 }
00161                 [request buildPostBody];
00162                 [self request:nil incrementUploadSizeBy:[request postLength]];
00163 
00164 
00165         } else {
00166                 [self request:nil incrementDownloadSizeBy:1];
00167                 [self request:nil incrementUploadSizeBy:1];
00168         }
00169         // Tell the request not to increment the upload size when it starts, as we've already added its length

00278 {
00279         // We handle certain methods differently because whether our delegate implements them or not can affect how the request should behave
00280 
00281         // If the delegate implements this, the request will stop to wait for credentials
00282         if (selector == @selector(authenticationNeededForRequest:)) {
00283                 if ([[self delegate] respondsToSelector:@selector(authenticationNeededForRequest:)]) {
00284                         return YES;
00285                 }
00286                 return NO;
00287 
00288         // If the delegate implements this, the request will to wait for credentials
00289         } else if (selector == @selector(proxyAuthenticationNeededForRequest:)) {
00290                 if ([[self delegate] respondsToSelector:@selector(proxyAuthenticationNeededForRequest:)]) {
00291                         return YES;
00292                 }
00293                 return NO;
00294 
00295         // If the delegate implements requestWillRedirectSelector, the request will stop to allow the delegate to change the url
00296         } else if (selector == @selector(request:willRedirectToURL:)) {
00297                 if ([self requestWillRedirectSelector] && [[self delegate] respondsToSelector:[self requestWillRedirectSelector]]) {
00298                         return YES;
00299                 }
00300                 return NO;
