00088 - (CGRect)frameForPagingScrollView {
00089   CGRect frame = self.bounds;
00090 
00091   // We make the paging scroll view a little bit wider on the side edges so that there
00092   // there is space between the pages when flipping through them.
00093   frame.origin.x -= self.pageHorizontalMargin;
00094   frame.size.width += (2 * self.pageHorizontalMargin);

00108 - (CGRect)frameForPageAtIndex:(NSInteger)pageIndex {
00109   // We have to use our paging scroll view's bounds, not frame, to calculate the page
00110   // placement. When the device is in landscape orientation, the frame will still be in
00111   // portrait because the pagingScrollView is the root view controller's view, so its
00112   // frame is in window coordinate space, which is never rotated. Its bounds, however,
00113   // will be in landscape because it has a rotation transform applied.
00114   CGRect bounds = _pagingScrollView.bounds;
00115   CGRect pageFrame = bounds;
00116 
00117   // We need to counter the extra spacing added to the paging scroll view in
00118   // frameForPagingScrollView:
00119   pageFrame.size.width -= self.pageHorizontalMargin * 2;
00120   pageFrame.origin.x = (bounds.size.width * pageIndex) + self.pageHorizontalMargin;
00121 
00122   return pageFrame;
00123 }
