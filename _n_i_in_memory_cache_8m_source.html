00228 - (void)storeObject:(id)object withName:(NSString *)name expiresAfter:(NSDate *)expirationDate {
00229   // Don't store nil objects in the cache.
00230   if (nil == object) {
00231     return;
00232   }
00233 
00234   if (nil != expirationDate && [[NSDate date] timeIntervalSinceDate:expirationDate] >= 0) {
00235     // The object being stored is already expired so remove the object from the cache altogether.
00236     [self removeObjectWithName:name];
00237 
00238     // We're done here.
00239     return;

00241   NIMemoryCacheInfo* info = [self cacheInfoForName:name];
00242 
00243   // Create a new cache entry.
00244   if (nil == info) {
00245     info = [[[NIMemoryCacheInfo alloc] init] autorelease];
00246     info.name = name;

00248 
00249   // Store the object in the cache item.
00250   info.object = object;

00350   NSDictionary* cacheMap = [self.cacheMap copy];

00352   // Iterate over the copied cache map (which will not be modified).

00480 - (void)didSetObject:(id)object withName:(NSString *)name {
00481   // Reduce the cache size after the object has been set in case the cache size is smaller
00482   // than the object that's being added and we need to remove this object right away. If we
00483   // try to reduce the cache size before the object's been set, we won't have anything to remove
00484   // and we'll get stuck in an infinite loop.
00485   if (self.maxNumberOfPixels > 0) {
00486     // Remove least recently used images until we satisfy our memory constraints.
00487     while (self.numberOfPixels > self.maxNumberOfPixels
00488            && [self.lruCacheObjects count] > 0) {
00489       NIMemoryCacheInfo* info = [self.lruCacheObjects firstObject];
00490       [self removeCacheInfoForName:info.name];
00491     }
