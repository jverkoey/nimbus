00031 int NIIsInDebugger(void) {
00032   int                 mib[4];
00033   struct kinfo_proc   info;
00034   size_t              size;
00035 
00036   // Initialize the flags so that, if sysctl fails for some bizarre
00037   // reason, we get a predictable result.
00038 
00039   info.kp_proc.p_flag = 0;
00040 
00041   // Initialize mib, which tells sysctl the info we want, in this case
00042   // we're looking for information about a specific process ID.
00043 
00044   mib[0] = CTL_KERN;
00045   mib[1] = KERN_PROC;
00046   mib[2] = KERN_PROC_PID;
00047   mib[3] = getpid();
00048 
00049   // Call sysctl.
00050 
00051   size = sizeof(info);
00052   sysctl(mib, sizeof(mib) / sizeof(*mib), &info, &size, NULL, 0);
00053 
00054   // We're being debugged if the P_TRACED flag is set.
00055 
00056   return (info.kp_proc.p_flag & P_TRACED) != 0;
00057 }
