00123 - (void)_setStatusBarHidden:(BOOL)hidden withAnimation:(UIStatusBarAnimation)animation {
00124   [self _setStatusBarHidden:hidden withAnimation:animation];
00125 
00126   if (UIStatusBarAnimationNone == animation) {
00127     [NIOverview view].alpha = 1;
00128     [NIOverview view].hidden = hidden;
00129 
00130   } else if (UIStatusBarAnimationSlide == animation) {
00131     [NIOverview view].alpha = 1;
00132 
00133     CGRect frame = [NIOverview frame];
00134 
00135     [UIView beginAnimations:nil context:nil];
00136     [UIView setAnimationDuration:NIStatusBarAnimationDuration()];
00137     [UIView setAnimationCurve:NIStatusBarAnimationCurve()];
00138 
00139     [NIOverview view].frame = frame;
00140 
00141     [UIView commitAnimations];
00142     
00143   } else if (UIStatusBarAnimationFade == animation) {
00144     CGRect frame = [NIOverview frame];
00145     [NIOverview view].frame = frame;
00146 
00147     [UIView beginAnimations:nil context:nil];
00148     [UIView setAnimationDuration:NIStatusBarAnimationDuration()];
00149     [UIView setAnimationCurve:NIStatusBarAnimationCurve()];
00150     
00151     [NIOverview view].alpha = hidden ? 0 : 1;
00152     
00153     [UIView commitAnimations];
