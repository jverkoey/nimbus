00297 - (void)setChromeVisibility:(BOOL)isVisible animated:(BOOL)animated {
00298   if (_isAnimatingChrome
00299       || (!isVisible && self.toolbar.hidden)
00300       || (isVisible && !self.toolbar.hidden)
00301       || !self.chromeCanBeHidden) {
00302     // Nothing to do here.
00303     return;
00304   }
00305 
00306   CGRect toolbarFrame = self.toolbar.frame;
00307   CGRect bounds = self.view.bounds;
00308 
00309   // Reset the toolbar's initial position.
00310   if (!isVisible) {
00311     toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
00312 
00313   } else {
00314     // Ensure that the toolbar is visible through the animation.
00315     self.toolbar.hidden = NO;
00316 
00317     toolbarFrame.origin.y = bounds.size.height;
00318   }
00319   self.toolbar.frame = toolbarFrame;
00320 
00321   // Show/hide the system chrome.
00322   if ([[UIApplication sharedApplication] respondsToSelector:
00323        @selector(setStatusBarHidden:withAnimation:)]) {
00324     // On 3.2 and higher we can slide the status bar out.
00325     [[UIApplication sharedApplication] setStatusBarHidden: !isVisible
00326                                             withAnimation: (animated
00327                                                             ? UIStatusBarAnimationSlide
00328                                                             : UIStatusBarAnimationNone)];
00329 
00330   } else {
00331     // On 3.0 devices we use the boring fade animation.
00332     [[UIApplication sharedApplication] setStatusBarHidden: !isVisible
00333                                                  animated: animated];
00334   }
00335 
00336   // Place the toolbar at its final location.
00337   if (isVisible) {
00338     // Slide up.
00339     toolbarFrame.origin.y = bounds.size.height - toolbarFrame.size.height;
00340 
00341   } else {
00342     // Slide down.
00343     toolbarFrame.origin.y = bounds.size.height;
00344   }
00345 
00346   // If there is a navigation bar, place it at its final location.
00347   CGRect navigationBarFrame = CGRectZero;
00348   if (nil != self.navigationController.navigationBar) {
00349     navigationBarFrame = self.navigationController.navigationBar.frame;
00350     CGRect statusBarFrame = [[UIApplication sharedApplication] statusBarFrame];
00351     CGFloat statusBarHeight = MIN(statusBarFrame.size.width, statusBarFrame.size.height);
00352 
00353     if (isVisible) {
00354       navigationBarFrame.origin.y = statusBarHeight;
00355 
00356     } else {
00357       navigationBarFrame.origin.y = 0;
00358     }
00359   }
00360 
00361   if (animated) {
00362     [UIView beginAnimations:nil context:nil];
00363     [UIView setAnimationDelegate:self];

00365                                          ? @selector(didShowChrome)

00367

00405 - (void)didTap {
00406   SEL selector = @selector(toggleChromeVisibility);
00407   if (self.photoAlbumView.zoomingIsEnabled) {
00408     // Cancel any previous delayed performs so that we don't stack them.
00409     [NSObject cancelPreviousPerformRequestsWithTarget:self selector:selector object:nil];
00410 
00411     // We need to delay taking action on the first tap in case a second tap comes in, causing

00416
