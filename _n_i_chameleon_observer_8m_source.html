00051   if ((self = [super init])) {
00052     // You must provide a stylesheet cache.
00053     NIDASSERT(nil != stylesheetCache);
00054     _stylesheetCache = [stylesheetCache retain];
00055     _stylesheetPaths = [[NSMutableArray alloc] init];
00056     _operations = [[NSOperationQueue alloc] init];
00057 
00058     if ([host hasSuffix:@"/"]) {
00059       _host = [host copy];
00060 
00061     } else {
00062       _host = [[host stringByAppendingString:@"/"] copy];
00063     }
00064 
00065     NSFileManager* fm = [NSFileManager defaultManager];
00066     NSDirectoryEnumerator* de = [fm enumeratorAtPath:_stylesheetCache.pathPrefix];
00067 
00068     NSString* filename;
00069     while ((filename = [de nextObject])) {
00070       BOOL isFolder = NO;
00071       NSString* path = [_stylesheetCache.pathPrefix stringByAppendingPathComponent:filename];
00072       if ([fm fileExistsAtPath:path isDirectory:&isFolder]
00073           && !isFolder
00074           && [[filename pathExtension] isEqualToString:@"css"]) {
00075         [_stylesheetPaths addObject:filename];
00076         NSString* cachePath = NIPathForDocumentsResource([self pathFromPath:filename]);
00077         NSError* error = nil;
00078         [fm removeItemAtPath:cachePath error:&error];
00079         [fm copyItemAtPath:path toPath:cachePath error:&error];
00080         NIDASSERT(nil == error);
00081       }
00082     }
00083   }

00119 - (void)operationWillFinish:(NINetworkRequestOperation *)operation {
00120   if (![operation.url.path isEqualToString:@"/watch"]) {
00121     NSMutableArray* changedStylesheets = [NSMutableArray array];
00122     NSArray* pathParts = [[operation.url absoluteString] pathComponents];
00123     NSString* path = [[pathParts subarrayWithRange:NSMakeRange(2, [pathParts count] - 2)]
00124                       componentsJoinedByString:@"/"];
00125     NSString* rootPath = NIPathForDocumentsResource(nil);
00126     NSString* hashedPath = [self pathFromPath:path];
00127     NSString* diskPath = [rootPath stringByAppendingPathComponent:hashedPath];
00128     [operation.data writeToFile:diskPath atomically:YES];
00129 
00130     NIStylesheet* stylesheet = [_stylesheetCache stylesheetWithPath:path loadFromDisk:NO];
00131     if ([stylesheet loadFromPath:path pathPrefix:rootPath delegate:self]) {
00132       [changedStylesheets addObject:stylesheet];
00133     }
00134     
00135     for (NSString* iteratingPath in _stylesheetPaths) {
00136       stylesheet = [_stylesheetCache stylesheetWithPath:iteratingPath loadFromDisk:NO];
00137       if ([stylesheet.dependencies containsObject:path]) {
00138         // This stylesheet has the changed stylesheet as a dependency so let's refresh it.
00139         if ([stylesheet loadFromPath:iteratingPath pathPrefix:rootPath delegate:self]) {
00140           [changedStylesheets addObject:stylesheet];
00141         }
00142       }
00143     }
00144 
00145     operation.processedObject = changedStylesheets;
00146   }
00147 }

00151 - (void)operationDidFinish:(NINetworkRequestOperation *)operation {
00152   if ([operation.url.path isEqualToString:@"/watch"]) {
