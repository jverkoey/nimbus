00089       NSString* key = [[kvPair objectAtIndex:0]
00090                        stringByReplacingPercentEscapesUsingEncoding:encoding];

00092       if (nil == values) {
00093         values = [NSMutableArray array];
00094         [pairs setObject:values forKey:key];
00095       }
00096       if (kvPair.count == 1) {
00097         [values addObject:[NSNull null]];
00098 
00099       } else if (kvPair.count == 2) {
00100         NSString* value = [[kvPair objectAtIndex:1]
00101                            stringByReplacingPercentEscapesUsingEncoding:encoding];
00102         [values addObject:value];
00103       }
00104     }

00183 - (NSComparisonResult)versionStringCompare:(NSString *)other {
00184   NSArray *oneComponents = [self componentsSeparatedByString:@"a"];
00185   NSArray *twoComponents = [other componentsSeparatedByString:@"a"];
00186 
00187   // The parts before the "a"
00188   NSString *oneMain = [oneComponents objectAtIndex:0];
00189   NSString *twoMain = [twoComponents objectAtIndex:0];
00190 
00191   // If main parts are different, return that result, regardless of alpha part
00192   NSComparisonResult mainDiff;
00193   if ((mainDiff = [oneMain compare:twoMain]) != NSOrderedSame) {
00194     return mainDiff;
00195   }
00196 
00197   // At this point the main parts are the same; just deal with alpha stuff
00198   // If one has an alpha part and the other doesn't, the one without is newer
00199   if ([oneComponents count] < [twoComponents count]) {
00200     return NSOrderedDescending;
00201 
00202   } else if ([oneComponents count] > [twoComponents count]) {
00203     return NSOrderedAscending;
00204 
00205   } else if ([oneComponents count] == 1) {
00206     // Neither has an alpha part, and we know the main parts are the same
00207     return NSOrderedSame;
00208   }
00209 
00210   // At this point the main parts are the same and both have alpha parts. Compare the alpha parts
00211   // numerically. If it's not a valid number (including empty string) it's treated as zero.
00212   NSNumber *oneAlpha = [NSNumber numberWithInt:[[oneComponents objectAtIndex:1] intValue]];
00213   NSNumber *twoAlpha = [NSNumber numberWithInt:[[twoComponents objectAtIndex:1] intValue]];
00214   return [oneAlpha compare:twoAlpha];
00215 }
