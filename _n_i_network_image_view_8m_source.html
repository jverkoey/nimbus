00142                             scaleOptions:(NINetworkImageViewScaleOptions)scaleOptions {

00144 
00145   NSString* cacheKey = cacheIdentifier;
00146 
00147   // Prefix cache key to create a namespace.
00148   if (nil != self.memoryCachePrefix) {
00149     cacheKey = [self.memoryCachePrefix stringByAppendingString:cacheKey];

00151 
00152   // Append the size to the key. This allows us to differentiate cache keys by image dimension.

00161   // (memoryCachePrefix)/path/to/image({width,height}{contentMode,cropImageForDisplay})

00329 - (void)setPathToNetworkImage:(NSString *)pathToNetworkImage forDisplaySize:(CGSize)displaySize contentMode:(UIViewContentMode)contentMode cropRect:(CGRect)cropRect {
00330   [self cancelOperation];
00331 
00332   if (NIIsStringWithAnyText(pathToNetworkImage)) {
00333     self.lastPathToNetworkImage = pathToNetworkImage;
00334 
00335     NSURL* url = nil;
00336 
00337     // Check for file URLs.
00338     if ([pathToNetworkImage hasPrefix:@"/"]) {
00339       // If the url starts with / then it's likely a file URL, so treat it accordingly.
00340       url = [NSURL fileURLWithPath:pathToNetworkImage];
00341 
00342     } else {
00343       // Otherwise we assume it's a regular URL.
00344       url = [NSURL URLWithString:pathToNetworkImage];
00345     }
00346 
00347     // If the URL failed to be created, there's not much we can do here.
00348     if (nil == url) {

00352     NINetworkImageRequest* request = [[[NINetworkImageRequest alloc] initWithURL:url] autorelease];

00359 - (void)setNetworkImageOperation:(NIOperation<NINetworkImageOperation> *)operation forDisplaySize:(CGSize)displaySize contentMode:(UIViewContentMode)contentMode cropRect:(CGRect)cropRect {
00360   [self cancelOperation];
00361 
00362   if (nil != operation) {
00363     // We explicitly do not allow negative display sizes. Check the call stack to figure
00364     // out who is providing a negative display size. It's possible that displaySize is an
00365     // uninitialized CGSize structure.
00366     NIDASSERT(displaySize.width >= 0);
00367     NIDASSERT(displaySize.height >= 0);
00368 
00369     // If an invalid display size is provided, use the image view's frame instead.
00370     if (0 >= displaySize.width || 0 >= displaySize.height) {
00371       displaySize = self.frame.size;
00372     }
00373 
00374     UIImage* image = nil;
00375 
00376     // Attempt to load the image from memory first.
00377     if (nil != self.imageMemoryCache) {
00378       NSString* cacheKey = [self cacheKeyForCacheIdentifier:operation.cacheIdentifier
00379                                                   imageSize:displaySize
00380                                                 contentMode:contentMode
00381                                                scaleOptions:self.scaleOptions];
00382       image = [self.imageMemoryCache objectWithName:cacheKey];
00383     }
00384 
00385     if (nil != image) {
00386       // We successfully loaded the image from memory.
00387       [self setImage:image];
00388 
00389       if ([self.delegate respondsToSelector:@selector(networkImageView:didLoadImage:)]) {
00390         [self.delegate networkImageView:self didLoadImage:self.image];
00391       }
00392 
00393     } else {
00394       // Unable to load the image from memory, so let's fire off the operation now.
00395       operation.delegate = self;
00396 
00397       operation.imageCropRect = cropRect;
00398       operation.scaleOptions = self.scaleOptions;
00399       operation.interpolationQuality = self.interpolationQuality;
00400       if (self.sizeForDisplay) {
00401         operation.imageDisplaySize = displaySize;
00402         operation.imageContentMode = contentMode;
00403       }
00404 
00405       self.operation = operation;
00406 
00407       [self.networkOperationQueue addOperation:self.operation];
00408     }
00409   }
