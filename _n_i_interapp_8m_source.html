00364 + (NSURL *)urlForInstagramImageAtFilePath:(NSString *)filePath error:(NSError **)error {
00365   UIImage* image = [[UIImage alloc] initWithContentsOfFile:filePath];
00366 
00367   // Unable to read the image.
00368   if (nil == image) {
00369     if (nil != error) {
00370       *error = [NSError errorWithDomain: NSCocoaErrorDomain
00371                                    code: NSFileReadUnknownError
00372                                userInfo: [NSDictionary dictionaryWithObject: filePath
00373                                                                      forKey: NSFilePathErrorKey]];
00374     }
00375     return nil;
00376   }
00377 
00378   // Instagram requires that images are at least 612x612 and preferably square.
00379   if (image.size.width < 612
00380       || image.size.height < 612) {
00381     if (nil != error) {
00382       *error = [NSError errorWithDomain: NINimbusErrorDomain
00383                                    code: NIImageTooSmall
00384                                userInfo: [NSDictionary dictionaryWithObject: image
00385                                                                      forKey: NIImageErrorKey]];
00386     }
00387     NI_RELEASE_SAFELY(image);
00388     return nil;
00389   }
00390 
00391   // Immediately remove the image from memory.
00392   NI_RELEASE_SAFELY(image);
00393   
00394   NSFileManager* fm = [NSFileManager defaultManager];
00395   NSArray* paths = NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES);
00396 
00397   NIDASSERT(NIIsArrayWithObjects(paths));
00398   if (!NIIsArrayWithObjects(paths)) {
00399     return nil;
00400   }
00401 
00402   NSString* documentsPath = [paths objectAtIndex:0];
00403   NSString* destinationPath = [documentsPath stringByAppendingPathComponent:
00404                                [NSString stringWithFormat:@"nimbus-instagram-image-%.0f.ig",
00405                                 [NSDate timeIntervalSinceReferenceDate]]];
00406 
00407   [fm copyItemAtPath: filePath
00408               toPath: destinationPath
00409                error: error];
00410 
00411   NIDASSERT(nil == error || nil == *error);
00412   if (nil == error || nil == *error) {
00413     return [NSURL URLWithString:[@"file:" stringByAppendingString:destinationPath]];
00414 
00415   } else {
