00254 - (void)layoutPages {
00255   if (nil == _scrollView || CGRectIsEmpty(_scrollView.frame)) {
00256     // Bail out early; the scroll view hasn't been laid out yet.
00257     return;
00258   }
00259 
00260   CGSize buttonDimensions = CGSizeZero;
00261   NSInteger numberOfRows = 0;
00262   NSInteger numberOfColumns = 0;
00263   CGFloat buttonHorizontalSpacing = 0;
00264   CGFloat buttonVerticalSpacing = 0;
00265   [self calculateLayoutForFrame: _scrollView.frame
00266                buttonDimensions: &buttonDimensions
00267                    numberOfRows: &numberOfRows
00268                 numberOfColumns: &numberOfColumns
00269         buttonHorizontalSpacing: &buttonHorizontalSpacing
00270           buttonVerticalSpacing: &buttonVerticalSpacing];
00271 
00272   NIDASSERT(numberOfRows > 0);
00273   NIDASSERT(numberOfColumns > 0);
00274 
00275   CGFloat pageWidth = _scrollView.frame.size.width;
00276 
00277   for (NSInteger ixPage = 0; ixPage < [_pagesOfButtons count]; ++ixPage) {
00278     CGFloat pageOffset = ixPage * pageWidth;
00279 
00280     NSArray* page = [_pagesOfButtons objectAtIndex:ixPage];
00281 
00282     CGFloat pageBottom = 0;
00283 
00284     for (NSInteger ixItem = 0; ixItem < [page count]; ++ixItem) {
00285       NSInteger col = ixItem % numberOfColumns;
00286       NSInteger row = ixItem / numberOfColumns;
00287 
00288       UIButton* button = [page objectAtIndex:ixItem];
00289       button.frame = CGRectMake(_padding.left + col * buttonDimensions.width
00290                                 + (col * buttonHorizontalSpacing),
00291                                 _padding.top + row * buttonDimensions.height
00292                                 + (row * buttonVerticalSpacing),
00293                                 buttonDimensions.width, buttonDimensions.height);
00294 
00295       pageBottom = MAX(pageBottom, CGRectGetMaxY(button.frame));
00296     }
00297 
00298     UIScrollView* pageScrollView = [_pagesOfScrollViews objectAtIndex:ixPage];
00299     pageScrollView.frame = CGRectMake(pageOffset, 0, pageWidth, _scrollView.frame.size.height);
00300     pageScrollView.contentSize = CGSizeMake(pageWidth, pageBottom + _padding.bottom);
00301   }

00431

00440   for (NSArray* page in _pagesOfButtons) {

00459                                   [self.dataSource launcherView: self
00460                                           numberOfButtonsInPage: ixPage]);
00461 
00462     NSMutableArray* page = [[[NSMutableArray alloc] initWithCapacity:numberOfItems]
00463                             autorelease];
00464 
00465     UIScrollView* pageScrollView = [[[UIScrollView alloc] init] autorelease];
00466     pageScrollView.indicatorStyle = UIScrollViewIndicatorStyleWhite;
00467 
00468     for (NSInteger ixItem = 0 ; ixItem < numberOfItems; ++ixItem) {
00469       UIButton* item = [self.dataSource launcherView: self
00470                                        buttonForPage: ixPage
00471                                              atIndex: ixItem];

00473                      action: @selector(didTapButton:)
