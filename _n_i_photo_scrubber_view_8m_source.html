00199 - (CGRect)frameForSelectionAtIndex:(NSInteger)photoIndex {
00200   CGSize photoSize = [self photoSize];
00201   CGSize selectionSize = [self selectionSize];
00202 
00203   CGFloat containerWidth = _containerView.bounds.size.width;
00204   // TODO (jverkoey July 21, 2011): I need to figure out why this is necessary.
00205   // Basically, when there are a lot of photos it seems like the selection frame
00206   // slowly gets offset from the thumbnail frame it's supposed to be representing until by the end
00207   // it's off the right edge by a noticeable amount. Trimming off some fat from the right
00208   // edge seems to fix this.
00209   if (_numberOfVisiblePhotos < _numberOfPhotos) {
00210     containerWidth -= photoSize.width / 2;
00211   }
00212 
00213   // Calculate the offset into the container view based on index/numberOfPhotos.
00214   CGFloat relativeOffset = floorf((((CGFloat)photoIndex * containerWidth)
00215                                    / (CGFloat)MAX(1, _numberOfPhotos)));
00216   
00217   return CGRectMake(floorf(_containerView.frame.origin.x
00218                            + relativeOffset
00219                            + photoSize.width / 2 - selectionSize.width / 2),
00220                     floorf(_containerView.center.y - selectionSize.height / 2),
00221                     selectionSize.width, selectionSize.height);

00238   [super layoutSubviews];
00239   
00240   CGSize boundsSize = self.bounds.size;
00241   
00242   CGSize photoSize = [self photoSize];
00243   CGFloat spaceBetweenPhotos = [self spaceBetweenPhotos];
00244   CGFloat maxContentWidth = [self maxContentWidth];
00245 
00246   // Update the total number of visible photos.
00247   _numberOfVisiblePhotos = [self numberOfVisiblePhotos];
00248 
00249   // Hide views if there isn't any interesting information to show.
00250   _containerView.hidden = (0 == _numberOfVisiblePhotos);
00251   _selectionView.hidden = (_selectedPhotoIndex < 0 || _containerView.hidden);
00252 
00253   // Calculate the container width using the number of visible photos.
00254   CGFloat containerWidth = ((_numberOfVisiblePhotos * photoSize.width)
00255                             + (MAX(0, _numberOfVisiblePhotos - 1) * spaceBetweenPhotos)
00256                             + _containerView.layer.borderWidth * 2);
00257 
00258   // Then we center the container in the content area.
00259   CGFloat containerMargins = MAX(0, floorf((maxContentWidth - containerWidth) / 2));
00260   CGFloat horizontalMargins = [self horizontalMargins];
00261   CGFloat containerHeight = photoSize.height + _containerView.layer.borderWidth * 2;
00262 
00263   CGFloat containerLeftMargin = horizontalMargins + containerMargins;
00264   CGFloat containerTopMargin = floorf((boundsSize.height - containerHeight) / 2);
00265 
00266   _containerView.frame = CGRectMake(containerLeftMargin,
00267                                     containerTopMargin,
00268                                     containerWidth,
00269                                     containerHeight);
00270 
00271   // Don't bother updating the selected photo index if there isn't a selection; the
00272   // selection view will be hidden anyway.
00273   if (_selectedPhotoIndex >= 0) {
00274     _selectionView.frame = [self frameForSelectionAtIndex:_selectedPhotoIndex];
00275   }
00276 
00277   // Update the frames for all of the thumbnails.

00293 - (void)updateVisiblePhotos {
00294   if (nil == self.dataSource) {
00295     return;
00296   }
00297 
00298   // This will update the number of visible photos if the layout did indeed change.
00299   [self layoutIfNeeded];
00300 
00301   // Recycle any views that we no longer need.
00302   while ([_visiblePhotoViews count] > _numberOfVisiblePhotos) {
00303     UIView* photoView = [_visiblePhotoViews lastObject];
00304     [photoView removeFromSuperview];
00305 
00306     [_recycledPhotoViews addObject:photoView];
00307 
00308     [_visiblePhotoViews removeLastObject];
00309   }
00310 
00311   // Lay out the visible photos.
00312   for (NSInteger ix = 0; ix < _numberOfVisiblePhotos; ++ix) {
00313     UIImageView* photoView = nil;
00314 
00315     // We must first get the photo view at this index.
00316 
00317     // If there aren't enough visible photo views then try to recycle another view.
00318     if (ix >= [_visiblePhotoViews count]) {
00319       photoView = [[[_recycledPhotoViews anyObject] retain] autorelease];
00320       if (nil == photoView) {
00321         // Couldn't recycle the view, so create a new one.
00322         photoView = [self photoView];
00323 
00324       } else {
00325         [_recycledPhotoViews removeObject:photoView];
00326       }
00327       [_containerView addSubview:photoView];
00328       [_visiblePhotoViews addObject:photoView];
00329 
00330     } else {
00331       photoView = [_visiblePhotoViews objectAtIndex:ix];
00332     }
00333 
00334     NSInteger photoIndex = [self photoIndexAtScrubberIndex:ix];
00335 
00336     // Only request the thumbnail if this thumbnail's photo index has changed. Otherwise
00337     // we assume that this photo either already has the thumbnail or it's still loading.
00338     if (photoView.tag != photoIndex) {
00339       photoView.tag = photoIndex;
00340 
00341       UIImage* image = [self.dataSource photoScrubberView:self thumbnailAtIndex:photoIndex];
00342       photoView.image = image;
00343 
00344       if (_selectedPhotoIndex == photoIndex) {
00345         _selectionView.image = image;
00346       }
00347     }
00348 
00349     photoView.frame = [self frameForThumbAtIndex:ix];
00350   }
00351 }

00361 - (NSInteger)photoIndexAtPoint:(CGPoint)point {
00362   NSInteger photoIndex;
00363   
00364   if (point.x <= 0) {
00365     // Beyond the left edge
00366     photoIndex = 0;

00482     animated = animated && (_selectedPhotoIndex >= 0);

00490       [UIView setAnimationBeginsFromCurrentState:YES];

00494     
00495     if (animated) {
00496       [UIView commitAnimations];

00500                                              thumbnailAtIndex: _selectedPhotoIndex];
