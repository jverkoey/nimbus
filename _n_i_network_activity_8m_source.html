00032 void NINetworkActivityTaskDidStart() {
00033   pthread_mutex_lock(&gMutex);
00034 
00035   BOOL enableNetworkActivityIndicator = (0 == gNetworkTaskCount);

00048 void NINetworkActivityTaskDidFinish() {
00049   pthread_mutex_lock(&gMutex);
00050 
00051   --gNetworkTaskCount;
00052   // If this asserts, you don't have enough stop requests to match your start requests.
00053   NIDASSERT(gNetworkTaskCount >= 0);
00054   gNetworkTaskCount = MAX(0, gNetworkTaskCount);
00055

00057     [UIApplication sharedApplication].networkActivityIndicatorVisible = NO;
00058   }
00059
